Bind null or undefined return value to Option

JavaScript

function getValue(key) {
  if (key) {
    return 'value';
  } else {
    return undefined;
  }
}
export { getValue };

Rescript

@module("./play") @return(nullable) external getValue: string => option<string> = "getValue"

/* usage */
let v = getValue("")
switch v {
  | Some(v') => Js.log(v')
  | None => Js.log("undefined")
}
